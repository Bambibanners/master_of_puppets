**Step B: Implement Docker-specific test flow in `run_stack_tests()` for runtime=="docker":**

```python
def run_stack_tests(stack_name, server_lxc, node_lxc, server_url, runtime, secrets, keep=False):
    results = []

    # CN-01: Server API reachable
    r_api = TestResult("CN-01: Docker server API reachable")
    if wait_for_server(server_url, timeout=SERVER_READY_TIMEOUT):
        r_api.ok(f"{server_url}/nodes → 200/401")
    else:
        r_api.fail(f"server not reachable after {SERVER_READY_TIMEOUT}s")
    results.append(r_api)
    if r_api.passed is False:
        # Skip remaining if server is down
        for name in ["CN-02..CN-08 (server unreachable)"]:
            results.append(TestResult(name).skip("server unreachable"))
        return results

    # Login and get JWT
    admin_password = secrets["ADMIN_PASSWORD"]
    jwt = api_login(server_url, admin_password)

    # Generate 2 JOIN_TOKENs (one per node — tokens are single-use)
    token_a = api_generate_token(server_url, jwt)
    time.sleep(2)
    token_b = api_generate_token(server_url, jwt)

    # Generate Ed25519 keypair for signing test jobs
    private_key = ed25519.Ed25519PrivateKey.generate()
    public_key_pem = private_key.public_key().public_bytes(
        serialization.Encoding.PEM,
        serialization.PublicFormat.SubjectPublicKeyInfo
    ).decode()
    upload_signing_key(server_url, jwt, public_key_pem)

    # CN-02: node-a enrollment
    r_a = TestResult("CN-02: Docker node-a enrolled + heartbeat")
    try:
        configure_insecure_registry_docker(node_lxc, f"{server_lxc_ip}:5000")
        enroll_node(node_lxc, server_url, token_a, "cross-net,node-a", "docker")
        # wait_for_n_heartbeats but just need node-a
        nodes = wait_for_n_heartbeats(server_url, jwt, 1, timeout=HEARTBEAT_TIMEOUT)
        if nodes:
            node_a = nodes[0]
            r_a.ok(f"node_id={node_a['node_id']}")
        else:
            r_a.fail("no heartbeat within timeout")
    except Exception as e:
        r_a.fail(str(e))
    results.append(r_a)

    # CN-03: node-b enrollment
    r_b = TestResult("CN-03: Docker node-b enrolled + heartbeat")
    try:
        enroll_node(node_lxc, server_url, token_b, "cross-net,node-b", "docker")
        nodes = wait_for_n_heartbeats(server_url, jwt, 2, timeout=HEARTBEAT_TIMEOUT)
        if len(nodes) >= 2:
            node_b = [n for n in nodes if n.get("node_id") != node_a.get("node_id", "")][0] if node_a else nodes[-1]
            r_b.ok(f"node_id={node_b['node_id']}")
        else:
            r_b.fail(f"only {len(nodes)} node(s) online, expected 2")
    except Exception as e:
        r_b.fail(str(e))
    results.append(r_b)

    # CN-04: Job execution cross-network
    r_exec = TestResult("CN-04: Docker job execution cross-network (COMPLETED + marker)")
    try:
        guid = dispatch_job(server_url, jwt, private_key, TEST_SCRIPT, ["cross-net"])
        job = poll_job_result(server_url, jwt, guid, timeout=JOB_POLL_TIMEOUT)
        if job.get("status") == "COMPLETED":
            output = job.get("output", "") or ""
            if TEST_SCRIPT_MARKER in output:
                r_exec.ok(f"status=COMPLETED, marker found, node={job.get('node_id','?')}")
            else:
                r_exec.fail(f"COMPLETED but marker '{TEST_SCRIPT_MARKER}' not in output")
        elif job:
            r_exec.fail(f"status={job.get('status','?')}")
        else:
            r_exec.fail(f"job did not complete within {JOB_POLL_TIMEOUT}s")
    except Exception as e:
        r_exec.fail(str(e))
    results.append(r_exec)

    # CN-05: Multi-node routing (job targeting node-a goes to node-a)
    r_route = TestResult("CN-05: Docker multi-node routing (job to node-a)")
    try:
        guid = dispatch_job(server_url, jwt, private_key, TEST_SCRIPT, ["node-a"])
        job = poll_job_result(server_url, jwt, guid, timeout=JOB_POLL_TIMEOUT)
        if job.get("status") == "COMPLETED":
            # Verify it ran on node-a (its node_id should match what we enrolled with node-a tag)
            r_route.ok(f"status=COMPLETED node={job.get('node_id','?')}")
        elif job:
            r_route.fail(f"status={job.get('status','?')}")
        else:
            r_route.fail("job did not complete")
    except Exception as e:
        r_route.fail(str(e))
    results.append(r_route)

    # CN-06: Image pull from LXC registry (enrollment success implies pull succeeded)
    r_img = TestResult("CN-06: Docker image pull from LXC registry")
    # This is verified indirectly: if node enrolled and heartbeated, it pulled the image
    # from the LXC-local registry (NODE_IMAGE points there)
    if r_a.passed or r_b.passed:
        r_img.ok("nodes enrolled using image from LXC-local registry (enrollment implies pull)")
    else:
        r_img.skip("nodes did not enroll — cannot verify image pull")
    results.append(r_img)

    # CN-07: Node revocation
    r_revoke = TestResult("CN-07: Docker node revocation (status=REVOKED)")
    try:
        all_nodes = api_get_nodes(server_url, jwt)
        # Revoke node-b (keep node-a for CN-08)
        node_to_revoke = next(
            (n for n in all_nodes if "node-b" in str(n.get("tags", ""))), None
        ) or (all_nodes[-1] if all_nodes else None)
        if node_to_revoke:
            revoked = revoke_node(server_url, jwt, node_to_revoke["node_id"])
            if revoked:
                time.sleep(5)
                updated = api_get_nodes(server_url, jwt)
                target = next((n for n in updated if n["node_id"] == node_to_revoke["node_id"]), None)
                if target and target.get("status") == "REVOKED":
                    r_revoke.ok(f"node_id={node_to_revoke['node_id']} status=REVOKED")
                else:
                    r_revoke.fail(f"status={target.get('status','?') if target else 'not found'}")
            else:
                r_revoke.fail("POST /nodes/{id}/revoke did not return 200")
        else:
            r_revoke.skip("no nodes found to revoke")
    except Exception as e:
        r_revoke.fail(str(e))
    results.append(r_revoke)

    # CN-08: Remaining node still active after revocation
    r_remain = TestResult("CN-08: Docker remaining node active post-revocation")
    try:
        guid = dispatch_job(server_url, jwt, private_key, TEST_SCRIPT, ["node-a"])
        job = poll_job_result(server_url, jwt, guid, timeout=JOB_POLL_TIMEOUT)
        if job.get("status") == "COMPLETED":
            r_remain.ok(f"non-revoked node completed job: {job.get('node_id','?')}")
        elif job:
            r_remain.fail(f"status={job.get('status','?')}")
        else:
            r_remain.fail("job did not complete")
    except Exception as e:
        r_remain.fail(str(e))
    results.append(r_remain)

    return results
```
