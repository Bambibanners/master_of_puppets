**ADDITION 3 — Mirror Source Settings panel:**

Add a new query and mutation for mirror-config at the top of SmelterRegistryManager (alongside existing queries):
```tsx
const [mirrorConfig, setMirrorConfig] = useState<{ pypi_mirror_url: string; apt_mirror_url: string } | null>(null);

const { data: mirrorConfigData } = useQuery({
    queryKey: ['mirror-config'],
    queryFn: async () => {
        const res = await authenticatedFetch('/api/admin/mirror-config');
        return res.json();
    },
    onSuccess: (data: any) => setMirrorConfig(data),
});

const updateMirrorConfigMutation = useMutation({
    mutationFn: async (payload: { pypi_mirror_url?: string; apt_mirror_url?: string }) => {
        const res = await authenticatedFetch('/api/admin/mirror-config', {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(payload),
        });
        return res.json();
    },
    onSuccess: () => {
        queryClient.invalidateQueries({ queryKey: ['mirror-config'] });
        toast.success('Mirror source settings saved');
    },
    onError: () => toast.error('Failed to save mirror settings'),
});
```

Note: `onSuccess` on useQuery requires react-query v4 callback syntax. If the project uses v5 (TanStack Query v5), use a `useEffect` to sync data instead:
```tsx
const { data: mirrorConfigData } = useQuery({ queryKey: ['mirror-config'], queryFn: ... });
// Then in render, use mirrorConfigData directly instead of local mirrorConfig state.
```
Check the existing query patterns in Admin.tsx (e.g. `const { data: config = ... } = useQuery(...)`) and follow the same pattern. Do not use `onSuccess` if the project uses TanStack Query v5.
