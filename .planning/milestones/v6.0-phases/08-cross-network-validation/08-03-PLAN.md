```python
def generate_gap_report(gaps: list, output_path: str) -> None:
    """Write Podman compatibility gap report to output_path."""
    import datetime
    lines = [
        "# Phase 8 — Podman Server Stack Compatibility Gaps",
        f"\n**Generated:** {datetime.date.today().isoformat()}",
        "\n## Summary\n",
        "Gaps found when running compose.server.yaml via podman-compose in an Ubuntu 24.04 LXC.",
        "These are deferred to a future Podman-parity phase — not fixed in Phase 8.\n",
        "\n## Known Gaps\n",
        "| # | Component | Issue | Severity | Workaround |\n|---|-----------|-------|----------|------------|",
    ]
    known = [
        ("1", "agent service", "/var/run/docker.sock mount fails in Podman LXC — Docker socket does not exist", "HIGH", "Remove docker.sock volume from compose.server.yaml for Podman deployments"),
        ("2", "db + agent startup", "depends_on: condition: service_healthy may not be respected by podman-compose < 1.1", "MEDIUM", "Add explicit sleep or retry in agent startup"),
    ]
    for row in known:
        lines.append(f"| {' | '.join(row)} |")
    if gaps:
        lines.append("\n## Runtime-Observed Gaps\n")
        for i, g in enumerate(gaps, 1):
            lines.append(f"{i}. {g}")
    else:
        lines.append("\n## Runtime-Observed Gaps\n\nNone detected beyond known gaps above.")
    lines.append("\n## Deferred To\n\nFuture phase: Podman-parity fix — adapt compose.server.yaml for Podman runtime.")

    os.makedirs(os.path.dirname(output_path), exist_ok=True)
    with open(output_path, "w") as f:
        f.write("\n".join(lines) + "\n")
    print(f"[GAP REPORT] Written to {output_path}")
```
