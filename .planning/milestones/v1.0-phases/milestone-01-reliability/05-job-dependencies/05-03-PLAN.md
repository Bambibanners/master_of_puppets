    ```python
    @staticmethod
    async def _unblock_dependents(completed_guid: str, db: AsyncSession):
        # 1. Find potential dependents
        # Using LIKE for JSON column search (simple for Phase 5)
        result = await db.execute(
            select(Job).where(
                Job.status == 'BLOCKED',
                Job.depends_on.like(f"%{completed_guid}%")
            )
        )
        dependents = result.scalars().all()
        
        for dep in dependents:
            try:
                upstreams_guids = json.loads(dep.depends_on)
                if completed_guid not in upstreams_guids:
                    continue
                
                # 2. Check all upstreams
                upstreams_res = await db.execute(select(Job).where(Job.guid.in_(upstreams_guids)))
                upstreams = upstreams_res.scalars().all()
                
                if all(u.status == "COMPLETED" for u in upstreams):
                    dep.status = "PENDING"
                    logger.info(f"🔓 Job {dep.guid} unblocked by completion of {completed_guid}")
            except:
                continue
    ```
