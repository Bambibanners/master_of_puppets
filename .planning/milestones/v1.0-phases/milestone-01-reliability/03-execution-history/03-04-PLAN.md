  <action>
    Create `puppeteer/dashboard/src/views/History.tsx`. This component should fetch data from `/api/executions` and display it in a table.

    ```tsx
    import React, { useState } from 'react';
    import { useQuery } from '@tanstack/react-query';
    import { authenticatedFetch } from '../auth';
    import { 
        Table, 
        TableBody, 
        TableCell, 
        TableHead, 
        TableHeader, 
        TableRow 
    } from '@/components/ui/table';
    import { Badge } from '@/components/ui/badge';
    import { Button } from '@/components/ui/button';
    import { formatDistanceToNow } from 'date-fns';
    import { Search, Filter, History as HistoryIcon, Terminal } from 'lucide-react';

    const History = () => {
        const [page, setPage] = useState(0);
        const limit = 25;

        const { data: executions, isLoading } = useQuery({
            queryKey: ['executions', page],
            queryFn: async () => {
                const res = await authenticatedFetch(`/api/executions?skip=${page * limit}&limit=${limit}`);
                return res.json();
            }
        });

        const getStatusVariant = (status: string) => {
            switch (status) {
                case 'COMPLETED': return 'default';
                case 'FAILED': return 'destructive';
                case 'SECURITY_REJECTED': return 'destructive';
                default: return 'outline';
            }
        };

        return (
            <div className="space-y-6">
                <div className="flex items-center justify-between">
                    <div>
                        <h2 className="text-3xl font-bold tracking-tight text-white">Execution History</h2>
                        <p className="text-zinc-500">Audit trail of all tasks across the network.</p>
                    </div>
                </div>

                <div className="rounded-xl border border-zinc-900 bg-zinc-975 overflow-hidden">
                    <Table>
                        <TableHeader className="bg-zinc-900/50">
                            <TableRow className="border-zinc-900">
                                <TableHead className="text-zinc-400">Timestamp</TableHead>
                                <TableHead className="text-zinc-400">Job GUID</TableHead>
                                <TableHead className="text-zinc-400">Node</TableHead>
                                <TableHead className="text-zinc-400">Status</TableHead>
                                <TableHead className="text-zinc-400 text-right">Duration</TableHead>
                                <TableHead className="text-zinc-400 text-right">Actions</TableHead>
                            </TableRow>
                        </TableHeader>
                        <TableBody>
                            {isLoading ? (
                                <TableRow><TableCell colSpan={6} className="text-center py-10 text-zinc-500">Loading history...</TableCell></TableRow>
                            ) : executions?.map((ex: any) => (
                                <TableRow key={ex.id} className="border-zinc-900 hover:bg-zinc-900/30 transition-colors">
                                    <TableCell className="font-medium text-zinc-300">
                                        {ex.started_at ? formatDistanceToNow(new Date(ex.started_at), { addSuffix: true }) : 'Pending'}
                                    </TableCell>
                                    <TableCell className="font-mono text-xs text-zinc-500">{ex.job_guid.slice(0, 8)}...</TableCell>
                                    <TableCell className="text-zinc-400">{ex.node_id || 'N/A'}</TableCell>
                                    <TableCell>
                                        <Badge variant={getStatusVariant(ex.status)}>{ex.status}</Badge>
                                    </TableCell>
                                    <TableCell className="text-right text-zinc-400">
                                        {ex.duration_seconds ? `${ex.duration_seconds.toFixed(1)}s` : '-'}
                                    </TableCell>
                                    <TableCell className="text-right">
                                        <Button variant="ghost" size="sm" className="gap-2 text-primary hover:text-primary hover:bg-primary/10">
                                            <Terminal className="h-3 w-3" />
                                            Logs
                                        </Button>
                                    </TableCell>
                                </TableRow>
                            ))}
                        </TableBody>
                    </Table>
                </div>
                
                <div className="flex items-center justify-end gap-2">
                    <Button 
                        variant="outline" 
                        size="sm" 
                        onClick={() => setPage(p => Math.max(0, p - 1))}
                        disabled={page === 0}
                        className="bg-zinc-900 border-zinc-800 text-zinc-400"
                    >
                        Previous
                    </Button>
                    <Button 
                        variant="outline" 
                        size="sm" 
                        onClick={() => setPage(p => p + 1)}
                        className="bg-zinc-900 border-zinc-800 text-zinc-400"
                    >
                        Next
                    </Button>
                </div>
            </div>
        );
    };

    export default History;
    ```
