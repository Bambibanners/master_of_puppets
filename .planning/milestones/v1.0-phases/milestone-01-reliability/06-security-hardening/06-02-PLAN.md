        # ... fetch job ...
        
        # 1. Fetch secrets for scrubbing
        decrypted_payload = decrypt_secrets(json.loads(job.payload))
        secrets_to_redact = []
        if "secrets" in decrypted_payload and isinstance(decrypted_payload["secrets"], dict):

        # 2. Scrub logs

        for entry in output_log:
